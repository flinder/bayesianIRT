model {
		for (j in 1 : J) {
			for (k in 1 : K) {
				p[j, k] <- 1 / (1 + exp( - exp(zeta[1, k]) * (theta[j] - zeta[2, k])));
				Y[j, k] ~ dbern(p[j, k]);
			}
		# Priors	
			theta[j] ~ dnorm(mu_t, tau_t);
		}
		for (k in 1:K) {
			zeta[, k] ~ dmnorm(mu_z, tau_z);
		}
		# Hyperpriors
		tau_t ~ dgamma(2, 2);
		mu_t ~ dnorm(0, tau_t);
		tau_z ~ dwish(sigma_0, 2)
		mu_z ~ dmnorm(mu_0, tau_z / 0.1);
	}
